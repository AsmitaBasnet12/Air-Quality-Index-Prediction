columns = ['PM2.5', 'PM10', 'NO', 'NO2', 'NOx', 'O3', 'CO', 'SO2']

# Process each column separately
for column in columns:
    Q1 = df1[column].quantile(0.25)
    Q3 = df1[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    # Replace outliers with Q1 or Q3
    df1[column] = df1[column].apply(
        lambda x: Q1 if x < lower_bound else (Q3 if x > upper_bound else x)
    )

# Defining hyperparameter ranges
C_values = [0.1, 1, 10, 100]
epsilon_values = [0.01, 0.1, 0.5]
kernel_types = ['linear', 'poly', 'rbf']

best_model = None
best_score = -np.inf
best_params = {}

# Iterate through all combinations of hyperparameters
for C in C_values:
    for epsilon in epsilon_values:
        for kernel in kernel_types:
            # Initialize and train the SVR model
            model = SVR(C=C, epsilon=epsilon, kernel=kernel)
            model.fit(X_train_scaled, y_train)

            # Predict the test set
            y_pred = model.predict(X_test_scaled)

            # Calculate R-squared score
            r2 = r2_score(y_test, y_pred)

            # If the current model is better, update the best model
            if r2 > best_score:
                best_model = model
                best_score = r2
                best_params = {'C': C, 'epsilon': epsilon, 'kernel': kernel}

# Print the best hyperparameters and corresponding score
print("Best Model Hyperparameters:")
print(f"C: {best_params['C']}")
print(f"Epsilon: {best_params['epsilon']}")
print(f"Kernel: {best_params['kernel']}")
print(f"Best R-squared Score: {best_score:.4f}")
